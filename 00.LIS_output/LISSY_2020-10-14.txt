. /*================================================== 
>               1: Program to calculate 400 bins 
> ==================================================*/ 
.  
. cap program drop _nq 

. discard 

. program define _nq, rclass 
  1.  
. syntax varname [fweight pweight/] /// 
>                 [if/] [in] [, /// 
>                 BY(varlist) /// 
>                 NQ(string)  /// 
>                 nvar(string) /// 
>                 ] 
  2.  
. qui { 
  3.  
.  // Weigths 
.  tempvar wgt 
  4.  if ("`weight'" == "") { 
  5.  gen `wgt' = 1 
  6.  } 
  7.  else { 
  8.  gen `wgt' = `exp' 
  9.  } 
 10.   
.  // calculations 
.   
.  tempvar a b c 
 11.   
.  sort `varlist' `wgt', stable 
 12.  
.  gen `a' = sum(`wgt')           // cummulative sum 
 13.  sum `wgt'                      // total number of people 
 14.  gen double `b' = `a'/`r(sum)'  // cummalitve share of population 
 15.  gen double `c' = `b'*`nq'      // rescale share of population by number of bins 
 16.  
.  gen long `nvar' = ceil(`c')    // round to ceil integer 
 17.  replace `nvar' = `nq' if `nvar' > `nq'  // replace to top of number of bins just in case 
 18.   
.  gen one = 1 
 19.   
.  noi table `nvar'  [`weight' = `wgt'], c(sum one mean `varlist'  min `varlist' max `varlist')  /*  
>   */ left concise format(%16.5f) 
 20.   
. } 
 21.  
. end 

.  
. /*================================================== 
>         2:  Loop over surveys 
> ==================================================*/ 
.  
.  
. foreach x of local surveys { 
  2.  cap { 
  3.  use ${`x'}, clear 
  4.   
.  local iso  = upper(iso3[1]) 
  5.  local year = year[1] 
  6.  local wave = wave[1] 
  7.  local currency : label currency `=currency[1]' 
  8.  
.  
.  keep hid  dhi hpopwgt nhhmem  
  9.  gen double popw=hpopwgt*nhhmem  
 10.  gen double lcu_pc = (dhi/nhhmem) // leave it annual per capita 
 11.  
.  drop if (lcu_pc < 0 | lcu_pc >= . | popw >= .) 
 12.  } 
 13.  if (_rc) continue 
 14.   
.  noi mata: printf("##1 `iso' `year' `wave'\n") 
 15.  noi mata: printf("##2 `currency'\n") 
 16.  _nq lcu_pc [pw = popw], nq(400) nvar(nq) 
 17. }
